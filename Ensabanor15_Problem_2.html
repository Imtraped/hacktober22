This code wants you to to take a string and check if it is an Anagram which is when a word/phrase is made using the same letters as a diffrent word or phase. The solution is gotten using if statements, lists, and sort methods to be solved. 


class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        # puts the variables into list form so that the characters can be moved around
        s = list(s)
        t = list(t)
        # sorts the list into alphabetical orders so they can be compared
        s.sort()
        t.sort()
        #checks if they are equal to each other and if they are then they
        if s == t:
            return True
            print ("true")
        else:
            return False
            print ("false")
